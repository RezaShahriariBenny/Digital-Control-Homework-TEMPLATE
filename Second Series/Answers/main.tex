    \begin{problem}{سوال اول}
    \raggedleft
    \lr{Solution:}
    
    $\frac{X(z)}{z} = \frac{2z^2+1}{(z-2)^2(z-1)} = \frac{9}{(z-2)^2} - \frac{1}{z-2} + \frac{3}{z-1}$
    
    \lr{then}
    
    $X(z) = \frac{9z^{-1}}{(1-2z^-1)^2} - \frac{1}{1-2z^-1} + \frac{3}{1-z^-1}$
    
    \lr{At last we have} :
    
    $x(k) = 9k(2^{k-1}) - 2^k + 3$
    
    
    \end{problem}
